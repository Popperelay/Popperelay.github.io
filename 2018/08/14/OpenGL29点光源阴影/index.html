上一篇文章中介绍的是定向光源的阴影，而这篇文章中我们将要介绍怎样生成点光源的阴影（Point Shadow，点阴影）。由于点光源的光线是向球面上所有方向发散的，所以点阴影也通常叫做万向阴影（omnidirectional shadow）。

在定向阴影中，由于光线方向是固定的，我们可以用一张2D贴图来保存从定向光源看过去的场景深度，点光源的光线方向不是固定的，而是朝向球面上的所有方向，那么我们应该怎么保存点光源视角下的场景深度呢？可以使用立方体贴图CubeMap来保存点光源视角下的整个场景的深度值。如下图所示：

生成深度立方体贴图

使用几何着色器的分层渲染

很容易想到的一个方案是仿照定向阴影的方式，把场景按6个方向渲染6次：用6个不同的观察矩阵，每次把一个不同的cubemap面附加到一个帧缓冲上，每次渲染一个面。类似下面这样：

for(int i = 0; i < 6; i++)
{
    GLuint face = GL_TEXTURE_CUBE_MAP_POSITIVE_X + i;
    glFramebufferTexture2D(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, face, depthCubemap, 0);
    BindViewMatrix(lightViewMatrices[i]);
    RenderScene();  
}

不过我们这里使用几何着色器的分层渲染技术：在几何着色器里通过使用gl_Layer内置变量，来指定渲染到cubemap的哪一个面，这样就能一遍pass同时渲染出cubemap的6个面，而不用分6遍pass把场景渲染6次。

生成深度cubemap的着色器

我们先来看一下着色器怎么写。

顶点着色器很简单：

#version 430 core

layout(location = 0) in vec3 _Position;

uniform mat4 u_Model;

void main()
{
    gl_Position = u_Model * vec4(_Position, 1.0);
}

几何着色器如下：

#version 430 core

layout(triangles) in;
layout(triangle_strip, max_vertices = 18) out;

uniform mat4 u_ShadowVPMatrices[6];

out vec4 g2f_WorldPos;

void main()
{
    for(int Face = 0; Face < 6; ++Face)
    {
        gl_Layer = Face;
        for(int i = 0; i < 3; ++i)
        {
            g2f_WorldPos = gl_in[i].gl_Position;
            gl_Position = u_ShadowVPMatrices[Face] * g2f_WorldPos;
            EmitVertex();
        }
        EndPrimitive();
    }
}

它需要从外界接收6个方向的观察和投影矩阵（从光源出发看向cubemap的6个面），设置几何着色器接收三角形图元，然后可以输出的最大顶点数量是18。在main函数里，对于每一个面：我们把输入的三角形的三个顶点，乘上光源视角下看向该面的观察和投影矩阵，生成一个新的三角形，这样我们就相当于我们以从光源出发看向6个面的不同的视角，把场景渲染了6次。然后配合几何着色器的内置变量gl_Layer，我们可以指定把当前生成的图元渲染到gl_Layer指定的面上（gl_Layer只在帧缓存附加了数组纹理或者cubemap时才有效）。

几何着色器所需要的6个观察和投影矩阵如下：

GLfloat ShadowNearPlane = 0.1f, ShadowFarPlane = 25.0f;
mat4 ShadowProjection = perspective(radians(90.0f), GLfloat(ShadowWidth) / ShadowHeight, ShadowNearPlane, ShadowFarPlane);
//从光源看向CubeMap六个面的投影矩阵*观察矩阵
std::vector<mat4> ShadowVPMatrics;
ShadowVPMatrics.push_back(ShadowProjection * lookAt(lightPos, lightPos + vec3(1, 0, 0), vec3(0, -1, 0)));
ShadowVPMatrics.push_back(ShadowProjection * lookAt(lightPos, lightPos + vec3(-1, 0, 0), vec3(0, -1, 0)));
ShadowVPMatrics.push_back(ShadowProjection * lookAt(lightPos, lightPos + vec3(0, 1, 0), vec3(0, 0, 1)));
ShadowVPMatrics.push_back(ShadowProjection * lookAt(lightPos, lightPos + vec3(0, -1, 0), vec3(0, 0, -1)));
ShadowVPMatrics.push_back(ShadowProjection * lookAt(lightPos, lightPos + vec3(0, 0, 1), vec3(0, -1, 0)));
ShadowVPMatrics.push_back(ShadowProjection * lookAt(lightPos, lightPos + vec3(0, 0, -1), vec3(0, -1, 0)));

片元着色器如下：

#version 430 core

in vec4 g2f_WorldPos;

uniform vec3 u_LightPos;
uniform float u_FarPlane;

void main()
{
    float Light2FragDistance = length(g2f_WorldPos.xyz - u_LightPos) / u_FarPlane;
    gl_FragDepth = Light2FragDistance;
}

我们把当前片元到光源的距离（世界坐标）写到帧缓存的Cubemap附件中，除以u_FarPlane是为了归一化到0到1之间。

当然，生成点阴影的深度cubemap，肯定需要帧缓存以及对应的cubemap纹理附件：

//生成立方体贴图
const GLuint ShadowWidth = 1024, ShadowHeight = 1024;
GLuint DepthCubeMap = -1;
glGenTextures(1, &DepthCubeMap);
glBindTexture(GL_TEXTURE_CUBE_MAP, DepthCubeMap);
for (int i = 0; i < 6; ++i)
    glTexImage2D(GL_TEXTURE_CUBE_MAP_POSITIVE_X + i, 0, GL_DEPTH_COMPONENT, ShadowWidth, ShadowHeight, 0, GL_DEPTH_COMPONENT, GL_FLOAT, nullptr);
glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_WRAP_R, GL_CLAMP_TO_EDGE);
glBindTexture(GL_TEXTURE_CUBE_MAP, 0);

//生成阴影贴图所需的FBO
GLuint ShadowFBO = -1;
glGenFramebuffers(1, &ShadowFBO);
glBindFramebuffer(GL_FRAMEBUFFER, ShadowFBO);
glFramebufferTexture(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, DepthCubeMap, 0);
glDrawBuffer(GL_NONE);
glReadBuffer(GL_NONE);
if (glCheckFramebufferStatus(GL_FRAMEBUFFER) != GL_FRAMEBUFFER_COMPLETE)
    std::cout << "Framebuffer not complete!" << std::endl;
glBindFramebuffer(GL_FRAMEBUFFER, 0);

再传入shader所需要的其他uniform变量，以及做好绘制场景所需要的操作以后，就能把场景的深度渲染到cubemap里。

显示深度cubemap

如果想要验证深度cubemap是否正确，因为我们无法把cubemap直观地显示出来（即使显示出来也不好看深度是否渲染正确），所以我们再绘制一次场景，但是片元的颜色是从cubemap中读出的深度值。显示cubemap深度所需的顶点着色器如下：

#version 430 core

layout(location = 0) in vec3 _Position;
layout(location = 1) in vec3 _Normal;
layout(location = 2) in vec2 _TexCoord;

out vec3 v2f_WorldPos;

uniform mat4 u_Projection;
uniform mat4 u_View;
uniform mat4 u_Model;

void main()
{
    gl_Position = u_Projection * u_View * u_Model * vec4(_Position, 1.0);
    v2f_WorldPos = vec3(u_Model * vec4(_Position, 1.0));
}

显示cubemap深度所需的片元着色器如下：

#version 430 core

in vec3 v2f_WorldPos;

out vec4 Color_;

uniform samplerCube u_DepthCubeMap;
uniform vec3 u_LightPos;

void main()
{
    vec3 Light2Frag = v2f_WorldPos - u_LightPos;
    float ClosestDepth = texture(u_DepthCubeMap, Light2Frag ).r;
    Color_ = vec4(vec3(ClosestDepth), 1.0);
}

在这个片元着色器里，使用samplerCube 类型的采样器，来从深度cubemap中采出场景的深度值，对应的采样坐标就是光源到片元的方向向量。

显示点阴影

有了深度cubemap以后，我们就可以渲染点光源照射下，场景里形成的点阴影了。

顶点着色器如下：

#version 430 core

layout(location = 0) in vec3 _Position;
layout(location = 1) in vec3 _Normal;
layout(location = 2) in vec2 _TexCoord;

out vec2 v2f_TexCoord;
out vec3 v2f_Normal;
out vec3 v2f_WorldPos;

uniform mat4 u_Projection;
uniform mat4 u_View;
uniform mat4 u_Model;
uniform bool u_IsReverseNormal;

void main()
{
    gl_Position = u_Projection * u_View * u_Model * vec4(_Position, 1.0);
    v2f_WorldPos = vec3(u_Model * vec4(_Position, 1.0));
    v2f_TexCoord = _TexCoord;
    if(u_IsReverseNormal)
        v2f_Normal = transpose(inverse(mat3(u_Model))) * (-1.0 * _Normal);
    else
        v2f_Normal = transpose(inverse(mat3(u_Model))) * _Normal;
}

其中u_IsReverseNormal，是想让最大的盒子的法线反相，这样点光源才能照亮最大盒子的内部的面。

片元着色器里使用blinn-phong光照模型，我们可以通过判断当前片元到点光源的实际距离，和从深度cubemap中以光源到当前片元的方向为采样坐标读出的最近距离，二者作比较，如果大于的话就说明当前片元处于阴影里：

float calculateShadow(vec3 vWorldPos)
{
    vec3 Light2Frag = vWorldPos - u_LightPos;
    float ClosestDepth = texture(u_DepthCubeMap, Light2Frag).r;
    ClosestDepth *= u_FarPlane;
    float CurrentDepth = length(Light2Frag);
    float Bias = 0.05;
    float Shadow = CurrentDepth - Bias > ClosestDepth ? 1.0 : 0.0;
    return Shadow;
}

函数传入的参数vWorldPos是当前片元的世界坐标。

完整的片元着色器如下：

#version 430 core

in vec2 v2f_TexCoord;
in vec3 v2f_Normal;
in vec3 v2f_WorldPos;

out vec4 Color_;

uniform sampler2D u_DiffuseTexture;
uniform samplerCube u_DepthCubeMap;
uniform vec3 u_LightPos;
uniform vec3 u_ViewPos;
uniform float u_FarPlane;
uniform vec3 u_LightColor = vec3(0.3);

float calculateShadow(vec3 vWorldPos)
{
    vec3 Light2Frag = vWorldPos - u_LightPos;
    float ClosestDepth = texture(u_DepthCubeMap, Light2Frag).r;
    ClosestDepth *= u_FarPlane;
    float CurrentDepth = length(Light2Frag);
    float Bias = 0.05;
    float Shadow = CurrentDepth - Bias > ClosestDepth ? 1.0 : 0.0;
    return Shadow;
}

void main()
{
    vec3 ObjectColor = texture(u_DiffuseTexture, v2f_TexCoord).rgb;
    vec3 Normal = normalize(v2f_Normal);
    //Ambient
    vec3 AmbientColor = 0.3 * ObjectColor;
    //Diffuse
    vec3 LightDirection = normalize(u_LightPos - v2f_WorldPos);
    float DiffuseFactor = max(dot(LightDirection, Normal), 0.0);
    vec3 DiffuseColor = DiffuseFactor * u_LightColor;
    //Specular
    vec3 ViewDirection = normalize(u_ViewPos - v2f_WorldPos);
    vec3 HalfDirection = normalize(LightDirection + ViewDirection);
    float SpecularFactor = pow(max(dot(Normal, HalfDirection), 0.0), 64.0);
    vec3 SpecularColor = SpecularFactor * u_LightColor;
    //Calculate Shadow
    float Shadow = calculateShadow(v2f_WorldPos);
    vec3 Result = AmbientColor + (1.0 - Shadow) * (DiffuseColor + SpecularColor) * ObjectColor;
    Color_ = vec4(Result, 1.0);
}

PCF

由于点阴影贴图基于传统阴影映射的原则，所以它也存在由于精度产生的反走样问题：靠近阴影边缘，就能看到明显的锯齿边。我们可以使用和上一篇文章一样的PCF技术，通过采样周围多个样本，来对阴影进行滤波。不同的只是上一篇文章是2D纹理，只需要在两个方向上采样即可，而我们这里对cubemap深度纹理，需要在三个方向上采样：

float calculateShadow(vec3 vWorldPos)
{
    vec3 Frag2Light = vWorldPos - u_LightPos;
    float CurrentDepth = length(Frag2Light);
    float Bias = 0.05;
    float Shadow = 0.0;
    float Samples = 4.0;
    float Offset = 0.1;
    float Step = Offset / (Samples * 0.5);
    for(float x = -Offset; x < Offset; x += Step)
    {
        for(float y = -Offset; y < Offset; y += Step)
        {
            for(float z = -Offset; z < Offset; z += Step)
            {
                float ClosestDepth = texture(u_DepthCubeMap, Frag2Light + vec3(x, y, z)).r;
                ClosestDepth *= u_FarPlane;
                if(CurrentDepth - Bias > ClosestDepth)
                    Shadow += 1.0;
            }
        }
    }
    Shadow /= (Samples * Samples * Samples);
    return Shadow;
}

代码应该很好理解，就是采样了当前片元在xyz三个方向上的周围64个点的深度值，然后和当前片元的实际深度值作比较，累加阴影值最后求均值。

改进的PCF

上面的PCF方面里，需要对每个片元采周围64个样本，太多了！而且其中大多数样本都是多余的，因为它们都是在原始方向向量的近处采样，这些采样点靠的太近，不够分散。不如在采样方向向量的垂直方向进行采样更有意义。

我们使用一个有着20个偏移方向的数组：

vec3 SampleOffsetDirections[20] = vec3[]
(
    vec3( 1,  1,  1), vec3( 1, -1,  1), vec3(-1, -1,  1), vec3(-1,  1,  1), 
    vec3( 1,  1, -1), vec3( 1, -1, -1), vec3(-1, -1, -1), vec3(-1,  1, -1),
    vec3( 1,  1,  0), vec3( 1, -1,  0), vec3(-1, -1,  0), vec3(-1,  1,  0),
    vec3( 1,  0,  1), vec3(-1,  0,  1), vec3( 1,  0, -1), vec3(-1,  0, -1),
    vec3( 0,  1,  1), vec3( 0, -1,  1), vec3( 0, -1, -1), vec3( 0,  1, -1)
);

然后使用这些偏移方向来采样：

float ViewDistance = length(u_ViewPos - v2f_WorldPos);
float DiskRadius = (1.0 + (ViewDistance / u_FarPlane)) / 25.0;
for(int i = 0; i < Samples; ++i)
{
    float ClosestDepth = texture(u_DepthCubeMap, Light2Frag + SampleOffsetDirections[i] * DiskRadius).r;
    ClosestDepth *= u_FarPlane;
    if(CurrentDepth - Bias > ClosestDepth)
        Shadow += 1.0;
}
Shadow /= float(Samples);

这样做的好处是与之前的PCF相比，达到相同效果所需要的采样数量变少了。

其中，DiskRadius 是偏移半径，可以是一个常数，也也可以像代码中那样，根据观察者到片元的距离来改变偏移半径：当距离更远的时候偏移半径较大，阴影更柔和；当距离更近的时候阴影就更锐利。
